	BIND( IRI(CONCAT("http://www.nationalarchives.gov.uk/PRONOM/",  ?puidLabel)) as ?compressionPuidIRI)

	BIND( IRI(CONCAT("http://data.nationalarchives.gov.uk/formatregistry/id/Actor/A-",  STR(?sourceActorId))) as ?sourceActorIRI)
